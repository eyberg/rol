def initialize(fn, modes)
@modes = reia_erl::r2e(modes.map{|m| reia_erl::r2e(m)})
@fn = fn.to_list()
case file::open(@fn, @modes)
when (~ok, file)
@file = file
when (~error, reason)
throw reason
end
end
